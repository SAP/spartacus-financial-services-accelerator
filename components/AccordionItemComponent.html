import { Component, Input } from '@angular/core';

@Component({
  selector: 'cx-fs-accordion-item',
  templateUrl: './accordion-item.component.html',
})
export class AccordionItemComponent {
  @Input() icon: string;
  @Input() opened = false;
  @Input() title: string;

  onItemClick() {
    this.opened = this.opened === true ? false : true;
  }
}

<div class="accordion-item">
  <h6 class="accordion-heading" (click)="onItemClick()" [class.active]="opened">
    <ng-container *ngIf="icon !== undefined">
      <span [ngClass]="[icon]"></span>
    </ng-container>
    {{ title }}
  </h6>
  <div class="accordion-item-wrapper" [class.active]="opened">
